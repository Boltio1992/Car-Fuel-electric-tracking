    <script>
        const today = new Date();
        const formattedDate = today.toISOString().split('T')[0];
        document.getElementById('driveDate').value = formattedDate;
        document.getElementById('energyDate').value = formattedDate;
        
        const monthNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];
        document.getElementById('dashMonthLabel').innerText = `Current Month: ${monthNames[today.getMonth()]} ${today.getFullYear()}`;

        const STORAGE_KEY = 'byd_final_logs';
        let logs = JSON.parse(localStorage.getItem(STORAGE_KEY)) || [];

        initApp();

        function initApp() {
            updateDashboard();
            updateTopStatus();
        }

        function showToast(message) {
            const toast = document.getElementById("toast");
            toast.innerText = message;
            toast.className = "toast show";
            setTimeout(function(){ toast.className = toast.className.replace("show", ""); }, 3000);
        }

        function calculateStats(dataset) {
            let dist=0, cost=0, kwh=0, trips=0, energy=0, evDist=0, hevDist=0;
            dataset.forEach(l => {
                let d = parseFloat(l.distance) || 0;
                dist += d; cost += parseFloat(l.costUSD) || 0; kwh += parseFloat(l.kwh) || 0;
                if (l.type.includes('Trip') && l.type !== 'Carryover') {
                    trips++;
                    if (l.type === 'EV Trip') evDist += d;
                    if (l.type === 'HEV Trip') hevDist += d;
                } else if (l.type === 'Charge' || l.type === 'Refuel') { energy++; }
            });
            return {dist, cost, kwh, trips, energy, evDist, hevDist};
        }

        function getTrendHTML(curr, prev, inverse = false) {
            if(prev === 0 && curr === 0) return '';
            if(prev === 0) return `<span style="font-size:0.75rem; color:#28a745; margin-left:6px;">(New)</span>`;
            const pct = ((curr - prev)/prev) * 100;
            if(pct === 0) return `<span style="font-size:0.75rem; color:var(--text-gray); margin-left:6px;">- 0%</span>`;
            
            const isUp = pct > 0;
            const isGood = inverse ? !isUp : isUp; 
            const color = isGood ? '#28a745' : '#dc3545'; 
            const arrow = isUp ? '↑' : '↓';
            return `<span style="font-size:0.75rem; color:${color}; margin-left:6px; font-weight:700;">${arrow} ${Math.abs(pct).toFixed(0)}%</span>`;
        }

        function updateDashboard() {
            const currMonthStr = today.getFullYear() + "-" + String(today.getMonth() + 1).padStart(2, '0');
            const prevDate = new Date(today.getFullYear(), today.getMonth() - 1, 1);
            const prevMonthStr = prevDate.getFullYear() + "-" + String(prevDate.getMonth() + 1).padStart(2, '0');

            let currLogs = logs.filter(l => l.date.startsWith(currMonthStr));
            let prevLogs = logs.filter(l => l.date.startsWith(prevMonthStr));

            let currStats = calculateStats(currLogs);
            let prevStats = calculateStats(prevLogs);

            document.getElementById('currDist').innerText = currStats.dist.toFixed(0);
            
            let costPerKm = currStats.dist > 0 ? (currStats.cost / currStats.dist) : 0;
            document.getElementById('insightCostKm').innerHTML = `$${costPerKm.toFixed(3)}<span> / km</span>`;
            document.getElementById('insightTrips').innerHTML = `${currStats.trips} ${getTrendHTML(currStats.trips, prevStats.trips)}`;
            document.getElementById('insightEvents').innerHTML = `${currStats.energy} ${getTrendHTML(currStats.energy, prevStats.energy)}`;

            document.getElementById('statCost').innerHTML = `$${currStats.cost.toFixed(2)} ${getTrendHTML(currStats.cost, prevStats.cost, true)}`;
            document.getElementById('statKwh').innerHTML = `${currStats.kwh.toFixed(1)} ${getTrendHTML(currStats.kwh, prevStats.kwh)}`;

            let evPct = currStats.dist > 0 ? Math.round((currStats.evDist / currStats.dist) * 100) : 0;
            let hevPct = currStats.dist > 0 ? Math.round((currStats.hevDist / currStats.dist) * 100) : 0;

            document.getElementById('evPct').innerText = evPct;
            document.getElementById('hevPct').innerText = hevPct;
            document.getElementById('evBar').style.width = evPct + "%";
            document.getElementById('hevBar').style.width = hevPct + "%";
        }

        function updateTopStatus() {
            const lastTrip = logs.slice().reverse().find(l => l.endSoc !== "" || l.endFuel !== "");
            if (lastTrip) {
                if (lastTrip.endSoc !== "") document.getElementById('topSoc').innerText = lastTrip.endSoc + "%";
                if (lastTrip.endFuel !== "") document.getElementById('topFuel').innerText = lastTrip.endFuel + "%";
            }
        }

        function openDrawer(id) {
            if(id === 'driveDrawer') {
                const lastTrip = logs.slice().reverse().find(l => l.endSoc !== "" || l.endFuel !== "");
                if (lastTrip) {
                    document.getElementById('driveStartSoc').value = lastTrip.endSoc;
                    document.getElementById('driveStartFuel').value = lastTrip.endFuel;
                }
            } else if (id === 'energyDrawer') {
                let defaultElec = 740;
                let defaultFuel = 4500;
                
                const lastCharge = logs.slice().reverse().find(l => l.type === 'Charge' && l.kwh > 0 && l.costKHR > 0);
                if (lastCharge) defaultElec = Math.round(lastCharge.costKHR / lastCharge.kwh);
                
                const lastRefuel = logs.slice().reverse().find(l => l.type === 'Refuel' && l.liters > 0 && l.costKHR > 0);
                if (lastRefuel) defaultFuel = Math.round(lastRefuel.costKHR / lastRefuel.liters);

                document.getElementById('rateElecInput').value = defaultElec;
                document.getElementById('rateFuelInput').value = defaultFuel;
            }
            document.getElementById(id).style.display = 'flex';
        }

        function closeDrawer(event, id) {
            if (event.target.className === 'overlay') document.getElementById(id).style.display = 'none';
        }

        function toggleEnergyFields() {
            const type = document.getElementById('energyType').value;
            document.getElementById('chargeInputs').style.display = (type === 'Charge') ? 'block' : 'none';
            document.getElementById('refuelInputs').style.display = (type === 'Refuel') ? 'block' : 'none';
            calcCost();
        }

        function estimateKwh() {
            let start = parseFloat(document.getElementById('chargeStart').value) || 0;
            let end = parseFloat(document.getElementById('chargeEnd').value) || 0;
            let capacity = parseFloat(document.getElementById('batteryCap').value) || 18.3;
            
            if (end > start) {
                let estimatedKwh = ((end - start) / 100) * capacity;
                document.getElementById('energyKwh').value = estimatedKwh.toFixed(2);
                calcCost();
            } else {
                alert("End SoC must be greater than Start SoC.");
            }
        }

        function calcCost() {
            const type = document.getElementById('energyType').value;
            let totalKHR = 0;

            if (type === 'Charge') {
                totalKHR = (parseFloat(document.getElementById('energyKwh').value) || 0) * (parseFloat(document.getElementById('rateElecInput').value) || 0);
            } else {
                totalKHR = (parseFloat(document.getElementById('energyLiters').value) || 0) * (parseFloat(document.getElementById('rateFuelInput').value) || 0);
            }

            if (totalKHR > 0) {
                document.getElementById('calcCostKHR').value = Math.round(totalKHR).toLocaleString() + " ៛";
                document.getElementById('calcCostUSD').value = "$" + (totalKHR / 4000).toFixed(2);
            } else {
                document.getElementById('calcCostKHR').value = ""; document.getElementById('calcCostUSD').value = "";
            }
        }

        function renderLedger() {
            const tbody = document.querySelector('#ledgerTable tbody');
            tbody.innerHTML = '';
            
            const monthFilter = document.getElementById('ledgerMonthFilter').value; 
            let displayLogs = logs;
            if (monthFilter) displayLogs = logs.filter(log => log.date.startsWith(monthFilter));

            if(displayLogs.length === 0) { tbody.innerHTML = '<tr><td colspan="5" style="text-align:center; padding:20px;">No logs found.</td></tr>'; return; }

            displayLogs.slice().reverse().forEach((log) => {
                const originalIdx = logs.indexOf(log);
                let val = log.distance > 0 ? `${log.distance}km` : (log.kwh ? `${log.kwh}kWh` : (log.liters ? `${log.liters}L` : '-'));
                let cost = log.costUSD > 0 ? `$${log.costUSD}` : '-';
                
                tbody.innerHTML += `
                    <tr>
                        <td style="font-size:0.75rem">${log.date.substring(5)}</td>
                        <td><strong>${log.type.split(' ')[0]}</strong><br><span style="font-size:0.7rem;opacity:0.7">${log.desc}</span></td>
                        <td>${val}</td>
                        <td style="font-weight:600">${cost}</td>
                        <td class="del-btn" onclick="deleteLog(${originalIdx})">✕</td>
                    </tr>
                `;
            });
        }

        function deleteLog(i) { if(confirm("Delete record?")) { logs.splice(i, 1); localStorage.setItem(STORAGE_KEY, JSON.stringify(logs)); initApp(); renderLedger(); showToast("Record Deleted"); } }
        function clearAllData() { if(confirm("Erase all data?")) { logs = []; localStorage.removeItem(STORAGE_KEY); initApp(); document.getElementById('ledgerDrawer').style.display='none'; showToast("All Data Cleared"); } }
        
        function exportCSV() {
            const monthFilter = document.getElementById('ledgerMonthFilter').value;
            let exportLogs = logs;
            if (monthFilter) exportLogs = logs.filter(log => log.date.startsWith(monthFilter));
            if(!exportLogs.length) { showToast("Logs empty."); return; }
            let csv = "Date,Type,Description,Distance (km),Start SoC (%),End SoC (%),Start Fuel (%),End Fuel (%),Charged (kWh),Refueled (L),Total Cost (USD)\n";
            exportLogs.forEach(r => { csv += `${r.date},"${r.type}","${r.desc}",${r.distance},${r.startSoc},${r.endSoc},${r.startFuel},${r.endFuel},${r.kwh},${r.liters},${r.costUSD}\n`; });
            const link = document.createElement("a"); link.href = URL.createObjectURL(new Blob([csv], { type: 'text/csv;charset=utf-8;' }));
            link.download = monthFilter ? `BYD_Logs_${monthFilter}.csv` : "BYD_Master_Logs.csv"; link.click();
            showToast("CSV Downloaded");
        }

        function backupData() {
            if(!logs.length) { showToast("No data to backup."); return; }
            const dataStr = JSON.stringify(logs);
            const blob = new Blob([dataStr], { type: "application/json" });
            const url = URL.createObjectURL(blob);
            const link = document.createElement("a");
            link.href = url;
            link.download = "BYD_System_Backup.json";
            link.click();
            showToast("Backup Downloaded");
        }

        function restoreData(event) {
            const file = event.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const importedLogs = JSON.parse(e.target.result);
                    if (Array.isArray(importedLogs)) {
                        logs = importedLogs;
                        localStorage.setItem(STORAGE_KEY, JSON.stringify(logs));
                        initApp();
                        document.getElementById('ledgerDrawer').style.display='none';
                        showToast("System Restored Successfully!");
                    } else {
                        alert("Invalid backup file format.");
                    }
                } catch (error) {
                    alert("Error reading backup file.");
                }
            };
            reader.readAsText(file);
        }
